fun fib(n) {
  if (n <= 1) return n;
  return fib(n - 2) + fib(n - 1);
}

for (var i = 0; i < 20; i = i + 1) {
  print fib(i);
}

var t_end = clock();
print "Time elapsed: " + toString(t_end - t_start) + " seconds.";
